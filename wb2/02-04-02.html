let canvas1 = /** @type {HTMLCanvasElement} */ (document.getElementById("canvas1"));
let context1 = canvas1.getContext('2d');

context1.save();
let gradient = context1.createRadialGradient(250, 250, 0, 250, 250, 200);
gradient.addColorStop(1, "red");
gradient.addColorStop(0, "rgba(255, 255, 255, 0)"); // transparency
context1.fillStyle = gradient;
context1.strokeStyle = "black";
context1.lineWidth = 10;
context1.beginPath();
context1.arc(250, 250, 200, 0, 2 * Math.PI, true); // big circle
context1.moveTo(250, 50);
context1.lineTo(250, 250);
context1.moveTo(250 - 200 / Math.sqrt(2), 250 + 200 / Math.sqrt(2));
context1.lineTo(250, 250);
context1.moveTo(250 + 200 / Math.sqrt(2), 250 + 200 / Math.sqrt(2));
context1.lineTo(250, 250);
context1.fill();
context1.stroke();
context1.restore();

context1.save(); // order
context1.fillStyle = "white";
context1.beginPath();
context1.arc(350, 200, 30, 0, 2 * Math.PI, true);
context1.arc(150, 200, 30, 0, 2 * Math.PI, true);
context1.fill();
context1.restore();
